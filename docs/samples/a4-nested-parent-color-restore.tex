\newtcolorbox{ParentColorBox}{
  enhanced,
  breakable,
  colback=blue!3,
  colframe=blue!70!black,
  colupper=blue!70!black,
  title={Parent colupper stays blue},
  before skip=6pt,
  after skip=6pt,
}

\newtcolorbox{ChildColorBox}{
  enhanced,
  breakable,
  colback=green!4,
  colframe=green!45!black,
  colupper=green!45!black,
  title={Nested child is green},
  before skip=6pt,
  after skip=6pt,
}

\newcommand{\filltext}[1]{%
  #1. This paragraph is intentionally ordinary text. It is repeated so that the
  nested breakable box has to split while the parent box continues afterwards.
  The parent text after the nested box must still use the parent color, not the
  nested child color and not the document default color.

}

nested color stack leaked into the parent text.

\filltext{PARENT-TAIL}
\end{ParentColorBox}
